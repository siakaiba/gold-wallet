function renderTable(data) { const tableBody = document.getElementById('goldTable'); tableBody.innerHTML = ''; if (!data) { updateStats(0, 0, 0, 0); return; } const entries = Object.keys(data).map(key => ({ id: key, ...data[key] })); // Filter for last 30 days and sort const thirtyDaysAgo = new Date(); thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30); const filteredEntries = entries .filter(e => e.timestamp >= thirtyDaysAgo.getTime()) .sort((a, b) => a.timestamp - b.timestamp); let currentWalletValue = 0, tGrams = 0, tInvested = 0; let cLabels = [], cData = []; const weights = { 24: 0, 22: 0, 21: 0 }; filteredEntries.forEach(e => { const liveRate = currentPrices.k24 * (e.purity / 24); const liveValOfEntry = Math.round(e.grams * liveRate); weights[e.purity] += e.grams; tGrams += e.grams; tInvested += e.totalCost; currentWalletValue += liveValOfEntry; cLabels.push(e.dateStr); cData.push(currentWalletValue); }); // Update Charts walletChart.data.labels = cLabels; walletChart.data.datasets[0].data = cData; walletChart.update('none'); barChart.data.datasets[0].data = [weights[24], weights[22], weights[21]]; barChart.update('none'); // Render Table Rows (New Column Logic) [...filteredEntries].reverse().forEach(e => { const currentClosingPrice = (currentPrices.k24 * (e.purity / 24)).toFixed(2); const profitPercent = (((currentClosingPrice * e.grams) - e.totalCost) / e.totalCost * 100).toFixed(2); tableBody.innerHTML += ` ${e.dateStr} ${e.dayName} QAR ${currentClosingPrice} ${profitPercent}% `; }); updateStats(filteredEntries.length, tGrams, Math.round(tInvested), currentWalletValue); }
