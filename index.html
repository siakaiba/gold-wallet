function updateStats(count, grams, invested, value) { const profit = value - invested; // Elements const valStat = document.getElementById('statValue'); const valCard = document.getElementById('cardValue'); const profitStat = document.getElementById('statProfit'); const profitCard = document.getElementById('cardProfit'); // Update Text document.getElementById('statCount').innerText = count; document.getElementById('statGrams').innerText = grams.toFixed(2) + 'g'; document.getElementById('statInvested').innerText = 'QAR ' + invested.toLocaleString(); valStat.innerText = 'QAR ' + value.toLocaleString(undefined, {maximumFractionDigits:0}); profitStat.innerText = 'QAR ' + profit.toLocaleString(undefined, {maximumFractionDigits: 0}); // Rule 1: Investment Value > Total Investment if (value > invested && invested > 0) { valStat.classList.replace('text-gray-800', 'text-green-600'); valCard.classList.add('bg-green-50', 'border-green-200'); } else { valStat.classList.replace('text-green-600', 'text-gray-800'); valCard.classList.remove('bg-green-50', 'border-green-200'); } // Rule 2: Net Profit/Loss > Total Investment if (profit > invested && invested > 0) { profitStat.classList.replace('text-gray-800', 'text-green-600'); profitCard.classList.add('bg-green-50', 'border-green-200'); } else { profitStat.classList.replace('text-green-600', 'text-gray-800'); profitCard.classList.remove('bg-green-50', 'border-green-200'); } }
